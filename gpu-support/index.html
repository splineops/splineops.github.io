GPU Support#

This guide demonstrates how to use SplineOps with CuPy for GPU interoperability.

GPU Interoperability Using CuPy#

This example demonstrates GPU interoperability using CuPy and tensor spline interpolation. The following steps are performed:

  1. Generate random data and coordinates.

  2. Create tensor splines using NumPy and CuPy.

  3. Evaluate the splines on a set of coordinates.

  4. Compute the absolute difference and mean square error between the NumPy and CuPy evaluations.

  5. Plot the results.

Imports#

1import numpy as np
2import matplotlib.pyplot as plt
3from splineops.interpolate.tensorspline import TensorSpline
4import cupy as cp

Data type#

1dtype = "float32"

Create random data samples and corresponding coordinates#

1nx, ny = 2, 5
2xmin, xmax = -3.1, +1
3ymin, ymax = 2, 6.5
4xx = np.linspace(xmin, xmax, nx, dtype=dtype)
5yy = np.linspace(ymin, ymax, ny, dtype=dtype)
6coordinates = xx, yy
7prng = np.random.default_rng(seed=5250)
8data = prng.standard_normal(size=tuple(c.size for c in coordinates))
9data = np.ascontiguousarray(data, dtype=dtype)

Tensor spline bases and modes#

1bases = "bspline3"
2modes = "mirror"

Create tensor spline from NumPy data#

1data_np = data
2coordinates_np = coordinates
3tensor_spline_np = TensorSpline(
4    data=data_np, coordinates=coordinates_np, bases=bases, modes=modes
5)

Create tensor spline from CuPy data for GPU computations#

1data_cp = cp.asarray(data)
2coordinates_cp = cp.asarray(xx), cp.asarray(yy)
3tensor_spline_cp = TensorSpline(
4    data=data_cp, coordinates=coordinates_cp, bases=bases, modes=modes
5)

Create evaluation coordinates (extended and oversampled in this case)#

1dx = (xx[-1] - xx[0]) / (nx - 1)
2dy = (yy[-1] - yy[0]) / (ny - 1)
3pad_fct = 1.1
4px = pad_fct * nx * dx
5py = pad_fct * ny * dy
6eval_xx = np.linspace(xx[0] - px, xx[-1] + px, 100 * nx)
7eval_yy = np.linspace(yy[0] - py, yy[-1] + py, 100 * ny)

Evaluate using NumPy#

1eval_coords_np = eval_xx, eval_yy
2data_eval_np = tensor_spline_np(coordinates=eval_coords_np)

Evaluate using CuPy#

1eval_coords_cp = cp.asarray(eval_xx), cp.asarray(eval_yy)
2data_eval_cp = tensor_spline_cp(coordinates=eval_coords_cp)

Compute difference#

1data_eval_cp_np = data_eval_cp.get()  # Convert CuPy array to NumPy array
2abs_diff = np.abs(data_eval_cp_np - data_eval_np)
3mse = np.mean((data_eval_cp_np - data_eval_np) ** 2)
4print(f"Maximum absolute difference: {np.max(abs_diff)}")
5print(f"Mean square error: {mse}")

The results on terminal:

Maximum absolute difference: 1.6689300537109375e-06
Mean square error: 1.372329577449885e-13

Plot results#

 1fig, axes = plt.subplots(
 2    nrows=1, ncols=3, sharex="all", sharey="all", layout="constrained"
 3)
 4ax = axes[0]
 5ax.imshow(data_eval_np.T)
 6ax.set_title("NumPy")
 7
 8data_eval_cp_np = data_eval_cp.get()  # Ensure we have a NumPy array
 9ax = axes[1]
10ax.set_title("CuPy")
11ax.imshow(data_eval_cp_np.T)
12
13ax = axes[2]
14ax.set_title(f"Absolute difference\n(MSE: {mse:.2e})")
15ax.imshow(abs_diff.T)
16
17plt.show()
